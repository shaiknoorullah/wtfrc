# wtfrc v0.1 "Ask" Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local-first CLI tool that indexes dotfiles/configs, stores them in a semantic SQLite knowledge base, and answers natural-language questions via a Bubble Tea popup REPL powered by a local LLM.

**Architecture:** Go binary with Cobra CLI, Viper config, modernc.org/sqlite for the knowledge base with FTS5, Bubble Tea TUI for the interactive REPL, and an LLM abstraction layer supporting Ollama and OpenAI-compatible APIs. The indexer parses configs via tool-specific parsers, enriches them with an LLM, and stores semantic entries. The responder searches the KB via FTS5, feeds matches to a fast local LLM, and streams the answer.

**Tech Stack:** Go 1.22+, Cobra, Viper, Bubble Tea, Lip Gloss, Bubbles, Glamour, modernc.org/sqlite, GoReleaser

**Spec:** `docs/design/2026-03-16-wtfrc-architecture.md`

```go
}
```

```go
	Schedule      string `mapstructure:"schedule"`

	v.SetDefault("indexer.watch", false)

	v.SetDefault("llm.strong.api_key_env", "ANTHROPIC_API_KEY")

}

// Load reads the config from the given path, or uses defaults if path is empty.
func Load(path string) (*Config, error) {
	v := viper.New()
	v.SetConfigType("toml")
	setDefaults(v)

	if path != "" {
		v.SetConfigFile(path)
		if err := v.ReadInConfig(); err != nil {
			return nil, err
		}
	}

	var cfg Config
	if err := v.Unmarshal(&cfg); err != nil {
		return nil, err
	}
	return &cfg, nil
}
```

```go
// internal/kb/db.go
package kb

import (
	"database/sql"
	"encoding/json"
	"fmt"
	"time"

	_ "modernc.org/sqlite"
)

type DB struct {
	conn *sql.DB
}

func Open(path string) (*DB, error) {
	conn, err := sql.Open("sqlite", path+"?_pragma=journal_mode(wal)&_pragma=foreign_keys(on)")
	if err != nil {
		return nil, fmt.Errorf("open db: %w", err)
	}
	db := &DB{conn: conn}
	if err := db.migrate(); err != nil {
		conn.Close()
		return nil, fmt.Errorf("migrate: %w", err)
	}
	return db, nil
}

func (db *DB) Close() error {
	return db.conn.Close()
}

func (db *DB) Conn() *sql.DB {
	return db.conn
}

func (db *DB) migrate() error {
	_, err := db.conn.Exec(schema)
	return err
}

const schema = `
CREATE TABLE IF NOT EXISTS entries (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  tool TEXT NOT NULL,
  type TEXT NOT NULL,
  raw_binding TEXT,
  raw_action TEXT,
  description TEXT NOT NULL,
  source_file TEXT NOT NULL,
  source_line INTEGER,
  category TEXT,
  see_also TEXT,
  indexed_at TEXT NOT NULL,
  file_hash TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS intents (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  entry_id INTEGER NOT NULL REFERENCES entries(id) ON DELETE CASCADE,
  phrase TEXT NOT NULL
);

CREATE VIRTUAL TABLE IF NOT EXISTS intents_fts USING fts5(phrase, content=intents, content_rowid=id);
CREATE VIRTUAL TABLE IF NOT EXISTS entries_fts USING fts5(description, content=entries, content_rowid=id);

CREATE TRIGGER IF NOT EXISTS intents_ai AFTER INSERT ON intents BEGIN
  INSERT INTO intents_fts(rowid, phrase) VALUES (new.id, new.phrase);
END;
CREATE TRIGGER IF NOT EXISTS intents_ad AFTER DELETE ON intents BEGIN
  INSERT INTO intents_fts(intents_fts, rowid, phrase) VALUES('delete', old.id, old.phrase);
END;
CREATE TRIGGER IF NOT EXISTS intents_au AFTER UPDATE ON intents BEGIN
  INSERT INTO intents_fts(intents_fts, rowid, phrase) VALUES('delete', old.id, old.phrase);
  INSERT INTO intents_fts(rowid, phrase) VALUES (new.id, new.phrase);
END;

CREATE TRIGGER IF NOT EXISTS entries_ai AFTER INSERT ON entries BEGIN
  INSERT INTO entries_fts(rowid, description) VALUES (new.id, new.description);
END;
CREATE TRIGGER IF NOT EXISTS entries_ad AFTER DELETE ON entries BEGIN
  INSERT INTO entries_fts(entries_fts, rowid, description) VALUES('delete', old.id, old.description);
END;
CREATE TRIGGER IF NOT EXISTS entries_au AFTER UPDATE ON entries BEGIN
  INSERT INTO entries_fts(entries_fts, rowid, description) VALUES('delete', old.id, old.description);
  INSERT INTO entries_fts(rowid, description) VALUES (new.id, new.description);
END;

CREATE TABLE IF NOT EXISTS manifest (
  file_path TEXT PRIMARY KEY,
  sha256 TEXT NOT NULL,
  mtime TEXT NOT NULL,
  last_indexed TEXT NOT NULL,
  entry_count INTEGER NOT NULL DEFAULT 0
);

CREATE TABLE IF NOT EXISTS sessions (
  id TEXT PRIMARY KEY,
  started_at TEXT NOT NULL,
  ended_at TEXT,
  query_count INTEGER DEFAULT 0,
  model_used TEXT,
  archived INTEGER DEFAULT 0
);

CREATE TABLE IF NOT EXISTS queries (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  session_id TEXT NOT NULL REFERENCES sessions(id),
  question TEXT NOT NULL,
  answer TEXT NOT NULL,
  entries_used TEXT,
  response_time_ms INTEGER,
  timestamp TEXT NOT NULL,
  accuracy_score REAL,
  issues TEXT
);

CREATE TABLE IF NOT EXISTS usage_events (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  tool TEXT NOT NULL,
  action_type TEXT NOT NULL DEFAULT '',
  action TEXT NOT NULL,
  optimal_action TEXT,
  entry_id INTEGER REFERENCES entries(id),
  timestamp TEXT NOT NULL,
  was_optimal INTEGER NOT NULL DEFAULT 0,
  coached INTEGER DEFAULT 0,
  time_saved_ms INTEGER
);

CREATE TABLE IF NOT EXISTS supervisor_runs (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  run_at TEXT NOT NULL,
  sessions_reviewed INTEGER,
  issues_found INTEGER,
  optimizations_applied TEXT,
  model_used TEXT
);

CREATE INDEX IF NOT EXISTS idx_entries_tool ON entries(tool);
CREATE INDEX IF NOT EXISTS idx_entries_type ON entries(type);
CREATE INDEX IF NOT EXISTS idx_intents_entry ON intents(entry_id);
CREATE INDEX IF NOT EXISTS idx_queries_session ON queries(session_id);
CREATE INDEX IF NOT EXISTS idx_usage_events_tool ON usage_events(tool);
CREATE INDEX IF NOT EXISTS idx_usage_events_ts ON usage_events(timestamp);
`

func (db *DB) InsertEntry(e KBEntry, intents []string) (int64, error) {
	seeAlsoJSON, _ := json.Marshal(e.SeeAlso)
	now := time.Now().UTC().Format(time.RFC3339)

	res, err := db.conn.Exec(
		`INSERT INTO entries (tool, type, raw_binding, raw_action, description, source_file, source_line, category, see_also, indexed_at, file_hash)
		 VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
		e.Tool, string(e.Type), e.RawBinding, e.RawAction, e.Description,
		e.SourceFile, e.SourceLine, e.Category, string(seeAlsoJSON), now, e.FileHash,
	)
	if err != nil {
		return 0, err
	}
	id, err := res.LastInsertId()
	if err != nil {
		return 0, err
	}
	for _, phrase := range intents {
		_, err := db.conn.Exec("INSERT INTO intents (entry_id, phrase) VALUES (?, ?)", id, phrase)
		if err != nil {
			return id, fmt.Errorf("insert intent: %w", err)
		}
	}
	return id, nil
}
```

```go
type SearchResult struct {

	Score float64

// Search queries both intents_fts and entries_fts, deduplicates, and returns
// the top N results ordered by relevance score.
func (db *DB) Search(query string, limit int) ([]SearchResult, error) {
	// Escape FTS5 special characters and build match expression
	q := ftsEscape(query)

	rows, err := db.conn.Query(`
		WITH intent_matches AS (
			SELECT i.entry_id, rank * -1.0 AS score
			FROM intents_fts f
			JOIN intents i ON i.id = f.rowid
			WHERE intents_fts MATCH ?
		),
		desc_matches AS (
			SELECT e.id AS entry_id, rank * -0.5 AS score
			FROM entries_fts f
			JOIN entries e ON e.id = f.rowid
			WHERE entries_fts MATCH ?
		),
		combined AS (
			SELECT entry_id, MAX(score) AS score FROM (
				SELECT * FROM intent_matches
				UNION ALL
				SELECT * FROM desc_matches
			) GROUP BY entry_id
		)
		SELECT e.id, e.tool, e.type, e.raw_binding, e.raw_action, e.description,
			   e.source_file, e.source_line, e.category, e.see_also, e.indexed_at, e.file_hash,
			   c.score
		FROM combined c
		JOIN entries e ON e.id = c.entry_id
		ORDER BY c.score DESC
		LIMIT ?
	`, q, q, limit)
	if err != nil {
		// FTS5 match can fail on empty or invalid query — return empty
		if strings.Contains(err.Error(), "fts5") {
			return nil, nil
		}
		return nil, fmt.Errorf("search: %w", err)
	}
	defer rows.Close()

	var results []SearchResult
	for rows.Next() {
		var r SearchResult
		var seeAlsoJSON, indexedAt string
		if err := rows.Scan(&r.ID, &r.Tool, &r.Type, &r.RawBinding, &r.RawAction,
			&r.Description, &r.SourceFile, &r.SourceLine, &r.Category,
			&seeAlsoJSON, &indexedAt, &r.FileHash, &r.Score); err != nil {
			return nil, err
		}
		r.IndexedAt, _ = time.Parse(time.RFC3339, indexedAt)
		if seeAlsoJSON != "" {
			json.Unmarshal([]byte(seeAlsoJSON), &r.SeeAlso)
		}
		// Load intents
		irows, err := db.conn.Query("SELECT phrase FROM intents WHERE entry_id = ?", r.ID)
		if err == nil {
			for irows.Next() {
				var p string
				irows.Scan(&p)
				r.Intents = append(r.Intents, p)
			}
			irows.Close()
		}
		results = append(results, r)
	}
	return results, nil
}

// ftsEscape wraps each word in double quotes so FTS5 treats them as literals.
func ftsEscape(query string) string {
	words := strings.Fields(query)
	for i, w := range words {
		words[i] = `"` + strings.ReplaceAll(w, `"`, `""`) + `"`
	}
	return strings.Join(words, " ")
}
```

```go
}

type I3Parser struct{}

func (p *I3Parser) Name() string { return "i3" }

func (p *I3Parser) CanParse(path string) bool {
	// Match i3/config or sway/config at any depth
	return strings.HasSuffix(path, "/i3/config") || strings.HasSuffix(path, "/sway/config")
}

var i3BindRe = regexp.MustCompile(`^\s*bind(?:sym|code)\s+(.+?)\s+(exec\s+.+|.+)$`)

func (p *I3Parser) Parse(path string) ([]RawEntry, error) {
	f, err := os.Open(path)
	if err != nil {
		return nil, err
	}
	defer f.Close()

	var entries []RawEntry
	scanner := bufio.NewScanner(f)
	lineNum := 0
	var prevLines []string

	for scanner.Scan() {
		lineNum++
		line := scanner.Text()
		prevLines = appendContext(prevLines, line, 3)

		trimmed := strings.TrimSpace(line)
		if strings.HasPrefix(trimmed, "#") || trimmed == "" {
			continue
		}

		matches := i3BindRe.FindStringSubmatch(trimmed)
		if matches == nil {
			continue
		}

		binding := matches[1]
		action := matches[2]

		entries = append(entries, RawEntry{
			Tool:         p.toolName(path),
			Type:         EntryKeybind,
			RawBinding:   binding,
			RawAction:    action,
			SourceFile:   path,
			SourceLine:   lineNum,
			ContextLines: strings.Join(prevLines, "\n"),
		})
	}
	return entries, scanner.Err()
}

func (p *I3Parser) toolName(path string) string {
	if strings.Contains(path, "/sway/") {
		return "sway"
	}
	return "i3"
}

func appendContext(lines []string, line string, maxLines int) []string {
	lines = append(lines, line)
	if len(lines) > maxLines {
		lines = lines[len(lines)-maxLines:]
	}
	return lines
}
```

```go
	}
```

```go
			})
			continue
		}
		// Aliases (double-quoted)
		if m := aliasReD.FindStringSubmatch(trimmed); m != nil {
			entries = append(entries, RawEntry{
				Tool: tool, Type: EntryAlias,

			})

		}
```

```go
// internal/indexer/parsers/kitty.go
package parsers

import (
	"bufio"
	"os"
	"regexp"
	"strings"
)

func init() {
	Register(&KittyParser{})
}

type KittyParser struct{}

func (p *KittyParser) Name() string { return "kitty" }

func (p *KittyParser) CanParse(path string) bool {
	return strings.HasSuffix(path, "kitty.conf") || strings.HasSuffix(path, "kitty/kitty.conf")
}

var kittyMapRe = regexp.MustCompile(`^\s*map\s+(\S+)\s+(.+)`)

func (p *KittyParser) Parse(path string) ([]RawEntry, error) {
	f, err := os.Open(path)
	if err != nil {
		return nil, err
	}
	defer f.Close()

	var entries []RawEntry
	scanner := bufio.NewScanner(f)
	lineNum := 0
	var prevLines []string

	for scanner.Scan() {
		lineNum++
		line := scanner.Text()
		prevLines = appendContext(prevLines, line, 3)
		trimmed := strings.TrimSpace(line)
		if trimmed == "" || strings.HasPrefix(trimmed, "#") {
			continue
		}

		if m := kittyMapRe.FindStringSubmatch(trimmed); m != nil {
			entries = append(entries, RawEntry{
				Tool:         "kitty",
				Type:         EntryKeybind,
				RawBinding:   m[1],
				RawAction:    m[2],
				SourceFile:   path,
				SourceLine:   lineNum,
				ContextLines: strings.Join(prevLines, "\n"),
			})
		}
	}
	return entries, scanner.Err()
}
```

```go
// internal/indexer/redactor.go
package indexer

import (
	"regexp"
	"strings"
)

type Redactor struct {
	patterns  []*regexp.Regexp
	sshKeyRe  *regexp.Regexp
	urlPassRe *regexp.Regexp
}

func NewRedactor(secretPatterns []string) *Redactor {
	var patterns []*regexp.Regexp
	for _, p := range secretPatterns {
		// Match the pattern followed by alphanumeric chars (a token/key value)
		re := regexp.MustCompile(`(?i)` + regexp.QuoteMeta(p) + `[A-Za-z0-9_\-]+`)
		patterns = append(patterns, re)
	}
	return &Redactor{
		patterns:  patterns,
		sshKeyRe:  regexp.MustCompile(`(?i)(IdentityFile\s+)\S+`),
		urlPassRe: regexp.MustCompile(`://(\w+):([^@]+)@`),
	}
}

func (r *Redactor) Redact(s string) string {
	// SSH key paths
	s = r.sshKeyRe.ReplaceAllString(s, "${1}[REDACTED_KEY_PATH]")

	// Passwords in URLs
	s = r.urlPassRe.ReplaceAllString(s, "://${1}:[REDACTED]@")

	// Pattern-based (API keys, tokens)
	for _, re := range r.patterns {
		s = re.ReplaceAllString(s, "[REDACTED]")
	}

	// Key=value with known secret patterns in the key name
	kvRe := regexp.MustCompile(`(?i)((?:password|secret|token|key)\s*=\s*)\S+`)
	s = kvRe.ReplaceAllString(s, "${1}[REDACTED]")

	return s
}

// RedactEntry applies redaction to all string fields of a RawEntry-like input.
func (r *Redactor) RedactFields(rawAction, contextLines *string) {
	if rawAction != nil {
		*rawAction = r.Redact(*rawAction)
	}
	if contextLines != nil {
		*contextLines = r.Redact(*contextLines)
	}
}
```

```go
// internal/llm/json.go
package llm

import (
	"context"
	"encoding/json"
	"fmt"
)

// CompleteJSON sends a JSON-mode completion and unmarshals the response.
// Retries once with a repair prompt on malformed JSON.
func CompleteJSON[T any](ctx context.Context, p Provider, req CompletionRequest) (T, error) {
	var zero T
	req.ResponseFormat = FormatJSON

	resp, err := p.Complete(ctx, req)
	if err != nil {
		return zero, err
	}

	var result T
	if err := json.Unmarshal([]byte(resp.Content), &result); err == nil {
		return result, nil
	}

	// Retry with repair prompt
	req.Messages = append(req.Messages, Message{Role: "assistant", Content: resp.Content})
	req.Messages = append(req.Messages, Message{
		Role:    "user",
		Content: "Your previous response was not valid JSON. Return ONLY valid JSON with no extra text.",
	})

	resp, err = p.Complete(ctx, req)
	if err != nil {
		return zero, fmt.Errorf("JSON repair retry failed: %w", err)
	}

	if err := json.Unmarshal([]byte(resp.Content), &result); err != nil {
		return zero, fmt.Errorf("LLM returned invalid JSON after retry: %w", err)
	}
	return result, nil
}
```

**Two-tier hallucination detection (§8.5):**

*Tier 1 — Deterministic (runs on every query):*

*Tier 2 — LLM cross-check (runs on ambiguous cases):*
- When deterministic check is inconclusive OR entries_used is empty but answer cites specific config values
- Send question + answer + KB entries to Strong LLM with the verification prompt from spec §8.5
- Parse JSON response: `{accurate: bool, hallucinated_refs: [...], contradictions: [...]}`
- Uses `CompleteJSON` helper for structured output
